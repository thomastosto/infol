CREATE TABLE Client(
   N_Client COUNTER,
   Nom VARCHAR(50),
   adresse VARCHAR(50),
   PRIMARY KEY(N_Client)
);

CREATE TABLE Commande(
   N_Commande COUNTER,
   DateComm DATE,
   PRIMARY KEY(N_Commande)
);

CREATE TABLE article(
   Référence_article COUNTER,
   libellé VARCHAR(50),
   Qté_Stock DOUBLE,
   PRIMARY KEY(Référence_article)
);

CREATE TABLE passer(
   N_Client INT,
   N_Commande INT,
   PRIMARY KEY(N_Client, N_Commande),
   FOREIGN KEY(N_Client) REFERENCES Client(N_Client),
   FOREIGN KEY(N_Commande) REFERENCES Commande(N_Commande)
);

CREATE TABLE concerner(
   N_Commande INT,
   Référence_article INT,
   Qté_Commandée DOUBLE,
   PRIMARY KEY(N_Commande, Référence_article),
   FOREIGN KEY(N_Commande) REFERENCES Commande(N_Commande),
   FOREIGN KEY(Référence_article) REFERENCES article(Référence_article)
);
